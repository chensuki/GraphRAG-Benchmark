## 概述

### 目标

将自己开发的RAG框架接入GraphRAG-Benchmark项目，实现：
- 标准化的数据加载和处理流程
- 统一的输出格式
- 公平的性能评估
- 可复现的实验结果

### 核心要求

1. **统一的数据输入格式**：使用项目提供的标准数据集
2. **标准化的输出格式**：确保评估脚本能够正确解析结果
3. **公平的实验条件**：使用相同的LLM、嵌入模型和参数设置
4. **完整的日志记录**：便于调试和复现

### 3. 数据格式说明

#### Corpus数据格式

**重要提示**：
- **Medical 数据集**：包含 1 个语料库（Medical）
- **Novel 数据集**：包含 **20 个独立的小说语料**，每个都需要独立索引
- 多语料库数据集需要特别注意并发控制，详见 [多语料库处理](#多语料库处理)

```json
[
  {
    "corpus_name": "Medical",           // 单语料库示例
    "context": "这里是完整的文本内容..."
  }
]
```

或

```json
[
  {"corpus_name": "Novel-58553", "context": "小说内容..."},
  {"corpus_name": "Novel-44557", "context": "另一部小说..."},
  // ... 共 20 个独立语料
]
```

#### Questions数据格式

```json
[
```

## 代码实现

### 1. 创建运行脚本

在`Examples/`目录下创建`run_my_framework.py`：

```python
"""
自定义框架集成示例
使用自定义RAG框架处理语料库并回答问题

功能：
1. 加载语料库数据
2. 使用自定义框架索引语料库
3. 查询问题并生成答案
4. 输出标准格式的预测结果（用于评估）
"""
import os
import sys
import asyncio
import argparse
import json
import logging
from pathlib import Path
from typing import Dict, List, Optional
from datasets import load_dataset
from tqdm import tqdm

# 配置日志
logging.basicConfig(
    format="%(asctime)s - %(levelname)s - %(message)s",
    level=logging.INFO
)
logger = logging.getLogger(__name__)

# 添加自定义框架路径（如果需要）
# framework_path = Path("/path/to/your/framework")
# if framework_path.exists():
#     sys.path.insert(0, str(framework_path))

# 导入自定义框架
try:
    from my_framework import MyRAGFramework
    logger.info("✅ 成功导入自定义框架")
except ImportError as e:
    logger.error(f"❌ 无法导入自定义框架: {e}")
    sys.exit(1)


def group_questions_by_source(question_list: List[dict]) -> Dict[str, List[dict]]:
    """按语料库名称分组问题"""
    grouped_questions = {}
    for question in question_list:
        source = question.get("source")
        if source not in grouped_questions:
            grouped_questions[source] = []
        grouped_questions[source].append(question)
    return grouped_questions


def process_corpus(
    corpus_name: str,
    context: str,
    base_dir: str,
    config_path: Optional[str],
    questions: Dict[str, List[dict]],
    sample: Optional[int],
    output_dir: str,
    top_k: int = 10,
    skip_build: bool = False
):
    """
    处理单个语料库：索引并回答问题
    
    Args:
        corpus_name: 语料库名称
        context: 语料库文本内容
        base_dir: 工作目录
        config_path: 配置文件路径
        questions: 按语料库名称分组的问题字典
        sample: 采样的问题数量
        output_dir: 输出目录
        top_k: 查询时返回的上下文数量
        skip_build: 是否跳过构建阶段
    """
    logger.info(f"📚 处理语料库: {corpus_name}")
    
    # 1. 准备输出路径
    os.makedirs(output_dir, exist_ok=True)
    output_path = os.path.join(output_dir, f"predictions_{corpus_name}.json")
    
    # 2. 初始化自定义框架
    try:
        # 根据您的框架初始化方式调整
        if config_path and os.path.exists(config_path):
            rag = MyRAGFramework(config_path=config_path)
        else:
            # 使用默认配置或参数初始化
            rag = MyRAGFramework(
                working_dir=os.path.join(base_dir, corpus_name),
                # 其他必要参数...
            )
        
        logger.info("✅ 自定义框架初始化成功")
    except Exception as e:
        logger.error(f"❌ 初始化失败: {e}")
        import traceback
        logger.debug(traceback.format_exc())
        return
    
    try:
        # 3. 索引语料库（如果未跳过构建）
        if not skip_build:
            logger.info(f"🔍 正在索引语料库: {corpus_name} ({len(context.split())} 词)")
            
            # 根据您的框架API调整索引方法
            rag.build_index(
                content=context,
                # 其他必要参数...
            )
            
            logger.info(f"✅ 索引完成: {corpus_name}")
        else:
            logger.info(f"⏭️  跳过构建阶段（假设语料库 {corpus_name} 已构建）")
        
        # 4. 获取该语料库的问题
        corpus_questions = questions.get(corpus_name, [])
        if not corpus_questions:
            logger.warning(f"⚠️ 未找到语料库 {corpus_name} 的问题")
            return
        
        # 5. 采样问题（如果指定）
        if sample and sample < len(corpus_questions):
            corpus_questions = corpus_questions[:sample]
            logger.info(f"🔍 采样了 {sample} 个问题（共 {len(questions.get(corpus_name, []))} 个）")
        
        logger.info(f"🔍 找到 {len(corpus_questions)} 个问题")
        
        # 6. 处理问题
        results = []
        for q in tqdm(corpus_questions, desc=f"回答问题 - {corpus_name}"):
            try:
                # 查询框架 - 根据您的框架API调整
                query_result = rag.query(
                    question=q["question"],
                    top_k=top_k,
                    # 其他必要参数...
                )
                
                # 提取结果 - 根据您的框架返回格式调整
                answer = query_result.get("answer", "")
                context_list = query_result.get("context", [])
                
                # 确保 context 是列表格式
                if isinstance(context_list, str):
                    context_list = [context_list] if context_list else []
                elif not isinstance(context_list, list):
                    context_list = []
                
                # 构建标准格式结果
                result = {
                    "id": q["id"],
                    "question": q["question"],
                    "source": corpus_name,
                    "context": context_list,  # 必须是列表
                    "evidence": q.get("evidence", ""),
                    "question_type": q.get("question_type", ""),
                    "generated_answer": answer,
                    "ground_truth": q.get("answer", "")
                }
                results.append(result)
                
            except Exception as e:
                logger.error(f"❌ 处理问题 {q.get('id')} 失败: {e}")
                results.append({
                    "id": q["id"],
                    "question": q["question"],
                    "source": corpus_name,
                    "context": [],
                    "evidence": q.get("evidence", ""),
                    "question_type": q.get("question_type", ""),
                    "generated_answer": f"查询失败: {str(e)}",
                    "ground_truth": q.get("answer", "")
                })
        
        # 7. 保存结果
        with open(output_path, "w", encoding="utf-8") as f:
            json.dump(results, f, indent=2, ensure_ascii=False)
        
        logger.info(f"💾 已保存 {len(results)} 个预测结果到: {output_path}")
        
    except Exception as e:
        logger.error(f"❌ 处理语料库 {corpus_name} 时出错: {e}")
        import traceback
        logger.debug(traceback.format_exc())
    finally:
        # 清理资源
        try:
            if hasattr(rag, 'close'):
                rag.close()
        except Exception as e:
            logger.warning(f"清理资源时出错: {e}")


def main():
    """主函数"""
    parser = argparse.ArgumentParser(
        description="自定义框架：处理语料库并回答问题",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog="""
示例用法:
  # 使用测试数据集（快速测试）
  python Examples/run_my_framework.py \\
    --subset medical_100 \\
    --config_path config/my_framework_config.yaml
  
  # 使用完整数据集
  python Examples/run_my_framework.py \\
    --subset medical \\
    --config_path config/my_framework_config.yaml
  
  # 跳过构建，直接查询（假设已构建）
  python Examples/run_my_framework.py \\
    --subset medical \\
    --config_path config/my_framework_config.yaml \\
    --skip-build
        """
    )
    
    # 核心参数
    parser.add_argument(
        "--subset",
        required=True,
        choices=["medical", "novel", "medical_100", "hotpotqa"],
        help="要处理的数据子集"
    )
    
    # 配置文件路径
    parser.add_argument(
        "--config_path",
        type=str,
        default=None,
        help="配置文件路径"
    )
    
    # 工作目录
    parser.add_argument(
        "--base_dir",
        default="./my_framework_workspace",
        help="工作目录"
    )
    
    # 采样和调试
    parser.add_argument(
        "--sample",
        type=int,
        default=None,
        help="每个语料库采样的问题数量（用于测试）"
    )
    
    # 查询配置
    parser.add_argument(
        "--top_k",
        type=int,
        default=10,
        help="查询时返回的上下文数量（默认 10）"
    )
    
    # 输出配置
    parser.add_argument(
        "--output_dir",
        default="./results/my_framework",
        help="输出目录"
    )
    
    # 构建/查询控制
    parser.add_argument(
        "--skip-build",
        action="store_true",
        help="跳过构建阶段，直接进入查询阶段（假设语料库已构建）"
    )
    
    args = parser.parse_args()
    
    # 定义数据路径（支持 Parquet 和 JSON 格式）
    SUBSET_PATHS = {
        "medical": {
            "corpus": "./Datasets/Corpus/medical.parquet",
            "questions": "./Datasets/Questions/medical_questions.parquet"
        },
        "novel": {
            "corpus": "./Datasets/Corpus/novel.parquet",
            "questions": "./Datasets/Questions/novel_questions.parquet"
        },
        "medical_100": {
            "corpus": "./Datasets/Corpus/medical.parquet",
            "questions": "./Datasets/Questions/medical_questions_100_balanced.json"
        },
        "hotpotqa": {
            "corpus": "./Datasets/Corpus/hotpotqa.parquet",
            "questions": "./Datasets/Questions/hotpotqa_questions.json"
        }
    }
    
    corpus_path = SUBSET_PATHS[args.subset]["corpus"]
    questions_path = SUBSET_PATHS[args.subset]["questions"]
    
    # 加载语料库数据（支持 Parquet 和 JSON 格式）
    try:
        if corpus_path.endswith('.json'):
            # 加载 JSON 格式
            with open(corpus_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
            if isinstance(data, dict):
                # 单个语料项（字典格式）
                corpus_data = [{
                    "corpus_name": data.get("corpus_name", "Unknown"),
                    "context": data.get("context", "")
                }]
            elif isinstance(data, list):
                # 多个语料项（列表格式）
                corpus_data = [{
                    "corpus_name": item.get("corpus_name", "Unknown"),
                    "context": item.get("context", "")
                } for item in data]
            else:
                raise ValueError(f"不支持的 JSON 格式: {type(data)}")
        else:
            # 加载 Parquet 格式
            corpus_dataset = load_dataset("parquet", data_files=corpus_path, split="train")
            corpus_data = []
            for item in corpus_dataset:
                corpus_data.append({
                    "corpus_name": item["corpus_name"],
                    "context": item["context"]
                })
        logger.info(f"📖 已加载 {len(corpus_data)} 个语料库文档")
    except Exception as e:
        logger.error(f"❌ 加载语料库失败: {e}")
        return
    
    # 采样语料库（如果指定）
    if args.sample:
        corpus_data = corpus_data[:1]
        logger.info(f"🔍 采样了 1 个语料库（用于测试）")
    
    # 加载问题数据（支持 Parquet 和 JSON 格式）
    try:
        if questions_path.endswith('.json'):
            # 加载 JSON 格式
            with open(questions_path, 'r', encoding='utf-8') as f:
                question_data = json.load(f)
            if not isinstance(question_data, list):
                raise ValueError(f"问题数据必须是列表格式，当前格式: {type(question_data)}")
            # 确保所有必需字段存在
            question_data = [{
                "id": item.get("id", ""),
                "source": item.get("source", ""),
                "question": item.get("question", ""),
                "answer": item.get("answer", ""),
                "question_type": item.get("question_type", ""),
                "evidence": item.get("evidence", "")
            } for item in question_data]
        else:
            # 加载 Parquet 格式
            questions_dataset = load_dataset("parquet", data_files=questions_path, split="train")
            question_data = []
            for item in questions_dataset:
                question_data.append({
                    "id": item["id"],
                    "source": item["source"],
                    "question": item["question"],
                    "answer": item["answer"],
                    "question_type": item["question_type"],
                    "evidence": item["evidence"]
                })
        grouped_questions = group_questions_by_source(question_data)
        logger.info(f"❓ 已加载 {len(question_data)} 个问题")
    except Exception as e:
        logger.error(f"❌ 加载问题失败: {e}")
        return
    
    # 处理每个语料库（带并发限制）
    async def process_all():
        # 重要：使用信号量限制并发数，避免 API 过载
        # Novel 数据集有 20 个语料，同时启动会导致 API 限流/超时
        max_concurrent = 2  # 建议值：1-3
        semaphore = asyncio.Semaphore(max_concurrent)
        
        async def process_with_limit(item):
            async with semaphore:
                await asyncio.to_thread(
                    process_corpus,
                    corpus_name=item["corpus_name"],
                    context=item["context"],
                    base_dir=args.base_dir,
                    config_path=args.config_path,
                    questions=grouped_questions,
                    sample=args.sample,
                    output_dir=args.output_dir,
                    top_k=args.top_k,
                    skip_build=args.skip_build
                )
        
        tasks = [process_with_limit(item) for item in corpus_data]
        await asyncio.gather(*tasks, return_exceptions=True)
    
    # 运行
    asyncio.run(process_all())
    logger.info("✅ 处理完成")


if __name__ == "__main__":
    main()
```

## 多语料库处理

### 问题说明

**Novel 数据集特殊性**：
- 包含 **20 个独立的小说语料**（Novel-58553, Novel-44557, ...）
- 每个语料都需要独立索引（调用 LLM 提取实体和关系）
- 同时启动所有索引任务会触发 API 限流或超时

### 解决方案

**方案 1：使用信号量限制并发**（推荐）

```python
async def process_multiple_corpora(corpus_data, max_concurrent=2):
    """处理多个语料库，带并发限制"""
    semaphore = asyncio.Semaphore(max_concurrent)
    
    async def process_with_limit(item):
        async with semaphore:
            # 索引和查询
            await process_corpus(item)
    
    tasks = [process_with_limit(item) for item in corpus_data]
    await asyncio.gather(*tasks, return_exceptions=True)
```

**方案 2：顺序处理**

```python
# 一个一个处理，最安全但最慢
for item in corpus_data:
    await process_corpus(item)
```

**方案 3：采样处理**（用于测试）

```bash
# 使用 --sample 1 只处理第一个语料
python Examples/run_my_framework.py --subset novel --sample 1
```

### 问题排查

如果遇到查询结果全部返回 `[no-context]`：

1. **检查索引状态**：查看 `workspace/Novel-XXXXX/` 目录
   - 如果缺少 `graph_chunk_entity_relation.graphml` 等文件 → 索引未完成
   - 如果 `kv_store_doc_status.json` 中状态为 `pending` → 索引被中断

2. **检查日志**：查找 API 超时或限流错误

3. **重新运行**：不带 `--skip-build` 参数重新索引

## 输出格式规范

### 标准输出格式

确保您的框架输出符合以下JSON格式：

```json
[
  {
    "id": "unique_question_id",
    "question": "问题文本",
    "source": "corpus_name",
    "context": [
      {"type": "chunk", "content": "文档片段内容..."},
      {"type": "entity", "content": "AlphaGo (artifact): AI系统描述...", "entity_name": "AlphaGo", "entity_type": "artifact", "description": "AI系统描述..."},
      {"type": "relationship", "content": "用户->订单: 创建了", "src_id": "用户", "tgt_id": "订单", "description": "创建了"}
    ],
    "evidence": "支持答案的证据文本",
    "question_type": "Fact Retrieval|Complex Reasoning|Contextual Summarize|Creative Generation",
    "generated_answer": "生成的答案文本",
    "ground_truth": "标准答案"
  }
]
```

### 统一 Context 格式说明

**格式结构**：

| type | content 格式 | 说明 |
|------|--------------|------|
| chunk | `"文档内容..."` | 原始文档片段 |
| entity | `"实体名 (类型): 描述"` | 结构化实体信息 |
| relationship | `"起点->终点: 描述"` | 结构化关系信息 |

**格式化工具函数**：

```python
from adapters.base import format_entity_content, format_relationship_content

# 格式化实体 content
content = format_entity_content("AlphaGo", "artifact", "AI系统描述...")
# 结果: "AlphaGo (artifact): AI系统描述..."

# 格式化关系 content
content = format_relationship_content("用户", "订单", "创建了")
# 结果: "用户->订单: 创建了"
```

### 与评估脚本的兼容性

**重要说明**：
- 框架输出使用统一格式：`context: List[Dict[str, Any]]`
- 评估脚本通过 `BenchmarkQueryResult.get_context_list()` 提取纯文本列表
- 实体和关系保留语义字段（entity_name, entity_type, src_id, tgt_id 等）

**评估兼容处理**：

```python
@dataclass
class BenchmarkQueryResult:
    answer: str
    context: List[Dict[str, Any]] = field(default_factory=list)

    def get_context_list(self) -> List[str]:
        """获取纯文本 context 列表（用于评估脚本）"""
        if not self.context:
            return []
        result = []
        for item in self.context:
            if isinstance(item, dict):
                result.append(item.get("content", ""))
        return [r for r in result if r]
```

### 字段说明

| 字段 | 类型 | 必需 | 说明 |
|------|------|------|------|
| id | string | ✓ | 问题的唯一标识符 |
| question | string | ✓ | 问题文本 |
| source | string | ✓ | 语料库名称 |
| context | array[dict] | ✓ | 统一格式的上下文列表，每个字典包含 `type` 和 `content` 字段 |
| evidence | string | ✓ | 支持答案的证据文本 |
| question_type | string | ✓ | 问题类型 |
| generated_answer | string | ✓ | 生成的答案 |
| ground_truth | string | ✓ | 标准答案 |

### 关键注意事项

1. **context字段必须是字典数组**：每个元素包含 `type` 和 `content` 字段
   ```python
   # ✅ 正确：统一格式
   "context": [
       {"type": "chunk", "content": "文档内容..."},
       {"type": "entity", "content": "AlphaGo (artifact): 描述", "entity_name": "AlphaGo", ...}
   ]

   # ❌ 错误：旧格式（纯字符串数组）
   "context": ["这是一个上下文片段"]
   ```

2. **使用格式化工具函数**：确保实体和关系的 content 字段格式正确
   ```python
   from adapters.base import format_entity_content, format_relationship_content

   # 格式化实体
   entity_content = format_entity_content("名称", "类型", "描述")
   # 结果: "名称 (类型): 描述"

   # 格式化关系
   rel_content = format_relationship_content("起点", "终点", "描述")
   # 结果: "起点->终点: 描述"
   ```

3. **字段类型必须匹配**：所有字符串字段都应该是string类型

```python
# 文本分块函数（保持一致）
def split_text(text: str, tokenizer, chunk_size=256, overlap=32):
    """标准文本分块方法"""
    tokens = tokenizer.encode(text, add_special_tokens=False)
    chunks = []
    start = 0
    while start < len(tokens):
        end = min(start + chunk_size, len(tokens))
        chunk_tokens = tokens[start:end]
        chunk_text = tokenizer.decode(chunk_tokens, skip_special_tokens=True)
        chunks.append(chunk_text)
        if end == len(tokens):
            break
        start += chunk_size - overlap
    return chunks
```

### Q1: 为什么 Novel 数据集查询结果全部返回 `[no-context]`？

**A**: 这通常是因为并发索引导致 API 超时，索引过程被中断。

**排查步骤**：
1. 检查工作目录：`workspace/Novel-XXXXX/kv_store_doc_status.json`
2. 如果状态为 `pending` → 索引未完成
3. 如果缺少图数据文件（如 `graph_chunk_entity_relation.graphml`）→ 索引失败

**解决方案**：
```python
# 使用信号量限制并发
semaphore = asyncio.Semaphore(2)  # 最多同时处理 2 个语料
```
