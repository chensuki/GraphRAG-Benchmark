## 数据返回规范

### 1. 查询结果对象

推荐定义一个标准的查询结果对象：

```python
from dataclasses import dataclass
from typing import List, Optional, Dict, Any

@dataclass
class BenchmarkQueryResult:
    """标准查询结果对象"""

    # 必需字段
    answer: str                                    # 生成的答案
    context: List[Dict[str, Any]] = field(default_factory=list)  # 统一格式的上下文列表

    def get_context_list(self) -> List[str]:
        """获取纯文本 context 列表（用于评估脚本）"""
        if not self.context:
            return []
        result = []
        for item in self.context:
            if isinstance(item, dict):
                result.append(item.get("content", ""))
        return [r for r in result if r]

    def get_chunks(self) -> List[Dict[str, Any]]:
        """获取所有 chunk 类型的 context"""
        return [item for item in self.context if isinstance(item, dict) and item.get("type") == "chunk"]

    def get_entities(self) -> List[Dict[str, Any]]:
        """获取所有 entity 类型的 context"""
        return [item for item in self.context if isinstance(item, dict) and item.get("type") == "entity"]

    def get_relationships(self) -> List[Dict[str, Any]]:
        """获取所有 relationship 类型的 context"""
        return [item for item in self.context if isinstance(item, dict) and item.get("type") == "relationship"]


def format_entity_content(name: str, entity_type: str, description: str) -> str:
    """格式化实体 content: "实体名 (类型): 描述" """
    if not name and not description:
        return ""
    parts = []
    if name:
        parts.append(f"{name} ({entity_type})" if entity_type else name)
    if description:
        parts.append(description)
    return ": ".join(parts)


def format_relationship_content(src: str, tgt: str, description: str) -> str:
    """格式化关系 content: "src->tgt: description" """
    if not src or not tgt:
        return ""
    return f"{src}->{tgt}: {description}" if description else f"{src}->{tgt}"
```

### 2. 上下文数据结构

#### 2.1 统一上下文格式（推荐）

```python
context = [
    {"type": "chunk", "content": "文档片段内容..."},
    {"type": "entity", "content": "AlphaGo (artifact): AI系统描述...", "entity_name": "AlphaGo", "entity_type": "artifact", "description": "AI系统描述..."},
    {"type": "relationship", "content": "用户->订单: 创建了", "src_id": "用户", "tgt_id": "订单", "description": "创建了"}
]
```

**格式说明**：

| type | content 格式 | 说明 |
|------|--------------|------|
| chunk | `"文档内容..."` | 原始文档片段 |
| entity | `"实体名 (类型): 描述"` | 结构化实体信息，保留语义字段 |
| relationship | `"起点->终点: 描述"` | 结构化关系信息，保留语义字段 |

**工具函数**（推荐使用）：

```python
from adapters.base import format_entity_content, format_relationship_content

# 格式化实体 content
content = format_entity_content("AlphaGo", "artifact", "AI系统描述...")
# 结果: "AlphaGo (artifact): AI系统描述..."

# 格式化关系 content
content = format_relationship_content("用户", "订单", "创建了")
# 结果: "用户->订单: 创建了"
```

**要求**：
- 必须是字典列表，每个字典包含 `type` 和 `content` 字段
- 实体和关系保留语义字段（entity_name, entity_type, description, src_id, tgt_id）
- 评估脚本通过 `get_context_list()` 方法提取纯文本列表

```python
# LightRAG的实现
from lightrag import LightRAG
from adapters.base import (
    BenchmarkQueryResult,
    format_entity_content,
    format_relationship_content
)

class LightRAGAdapter:
    def __init__(self, working_dir, llm_model_func, embedding_func, **kwargs):
        self.rag = LightRAG(
            working_dir=working_dir,
            llm_model_func=llm_model_func,
            embedding_func=embedding_func,
            **kwargs
        )

    async def build_index(self, content, **kwargs):
        """构建索引"""
        await self.rag.ainsert(content)

    async def aquery(self, question, top_k=10, **kwargs):
        """查询并返回答案"""
        from lightrag import QueryParam
        param = QueryParam(mode="hybrid", top_k=top_k)
        result = await self.rag.aquery(question, param=param)
        return result

    async def aquery_full(self, question, top_k=10, **kwargs):
        """完整查询，返回统一格式"""
        from lightrag import QueryParam
        param = QueryParam(mode="hybrid", top_k=top_k)

        # 使用 aquery_llm 获取答案和结构化数据
        result = await self.rag.aquery_llm(question, param=param)

        # 提取 LLM 响应
        llm_response = result.get("llm_response", {})
        answer = llm_response.get("content", "")

        # 构建统一格式的 context
        context = self._build_unified_context(result)

        return BenchmarkQueryResult(answer=answer, context=context)

    def _build_unified_context(self, result: dict) -> List[dict]:
        """构建统一格式的 context"""
        context = []
        if not result or result.get("status") == "failure":
            return context

        data = result.get("data", {})
        if not data:
            return context

        # 提取 chunks
        for chunk in data.get("chunks", []):
            content = chunk.get("content", "") if isinstance(chunk, dict) else str(chunk)
            if content:
                context.append({"type": "chunk", "content": content})

        # 提取 entities
        for entity in data.get("entities", []):
            if isinstance(entity, dict):
                entity_name = entity.get("entity_name", "")
                entity_type = entity.get("entity_type", "")
                description = entity.get("description", "")
                content = format_entity_content(entity_name, entity_type, description)
                if content:
                    context.append({
                        "type": "entity",
                        "content": content,
                        "entity_name": entity_name,
                        "entity_type": entity_type,
                        "description": description
                    })

        # 提取 relationships
        for rel in data.get("relationships", []):
            if isinstance(rel, dict):
                src = rel.get("src_id", "")
                tgt = rel.get("tgt_id", "")
                desc = rel.get("description", "")
                content = format_relationship_content(src, tgt, desc)
                if content:
                    context.append({
                        "type": "relationship",
                        "content": content,
                        "src_id": src,
                        "tgt_id": tgt,
                        "description": desc
                    })

        return context
```

## 接口实现最佳实践

### 1. 错误处理

**关键原则**：不抛出未捕获的异常，总是返回有效的结果

```python
async def aquery_full(self, query, **kwargs):
    """完整查询，带错误处理"""
    try:
        # 执行查询
        result = await self._internal_query(query, **kwargs)
        
        # 确保返回格式正确
        if result is None:
            return QueryResult(
                answer="查询失败：无返回结果",
                context=[]
            )
        
        # 灵活处理上下文格式（与评估脚本一致）
        context = result.get("context", [])
        if isinstance(context, str):
            context = [context] if context else []
        elif not isinstance(context, list):
            context = []
        
        # 确保答案不为空
        answer = result.get("answer", "")
        if answer is None:
            answer = "无法生成答案"
        
        return QueryResult(
            answer=str(answer),
            context=context
        )
        
    except Exception as e:
        logger.error(f"查询失败: {e}")
        # 返回安全的默认结果，而不是抛出异常
        return QueryResult(
            answer=f"查询失败: {str(e)}",
            context=[]
        )
```

**最佳实践**：
1. **捕获所有异常**：在接口层面捕获所有可能的异常
2. **返回安全默认值**：失败时返回空上下文和错误消息作为答案
3. **记录日志**：记录详细的错误信息用于调试
4. **不中断流程**：确保单个查询失败不影响批量处理

### 5. 结果缓存（可选）

```python
from functools import lru_cache
import hashlib

def get_cache_key(question: str, **kwargs) -> str:
    """生成缓存键"""
    key_data = f"{question}_{sorted(kwargs.items())}"
    return hashlib.md5(key_data.encode()).hexdigest()

class CachedRAGFramework:
    def __init__(self, rag_instance, cache_size=100):
        self.rag = rag_instance
        self.cache = {}
        self.cache_size = cache_size
    
    async def aquery(self, question, **kwargs):
        """带缓存的查询"""
        cache_key = get_cache_key(question, **kwargs)
        
        if cache_key in self.cache:
            logger.debug(f"使用缓存结果: {cache_key}")
            return self.cache[cache_key]
        
        result = await self.rag.aquery(question, **kwargs)
        
        # 添加到缓存
        if len(self.cache) >= self.cache_size:
            # 简单的LRU：删除第一个
            first_key = next(iter(self.cache))
            del self.cache[first_key]
        
        self.cache[cache_key] = result
        return result
```

### 6. 批量查询支持

```python
async def abatch_query(
    self, 
    questions: List[str], 
    top_k: int = 10,
    **kwargs
) -> List[QueryResult]:
    """
    批量查询
    
    Args:
        questions: 问题列表
        top_k: 每个问题返回的上下文数量
        **kwargs: 其他参数
        
    Returns:
        查询结果列表
    """
    tasks = [self.aquery(q, top_k=top_k, **kwargs) for q in questions]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # 处理异常
    processed_results = []
    for i, result in enumerate(results):
        if isinstance(result, Exception):
            logger.error(f"问题 {i} 查询失败: {result}")
            processed_results.append(QueryResult(
                answer=f"查询失败: {str(result)}",
                context=[]
            ))
        else:
            processed_results.append(result)
    
    return processed_results
```

### 7. 多语料库处理（重要）

**问题**：某些数据集（如 Novel）包含多个独立语料库，每个需要独立索引。

**示例**：Novel 数据集有 20 个独立的小说语料
```python
corpus_data = [
    {"corpus_name": "Novel-58553", "context": "小说内容..."},
    {"corpus_name": "Novel-44557", "context": "另一部小说..."},
    # ... 共 20 个
]
```

**正确做法**：使用信号量限制并发，避免 API 过载

```python
async def process_multiple_corpora(corpus_data, questions, max_concurrent=2):
    """
    处理多个语料库
    
    Args:
        corpus_data: 语料库列表
        questions: 分组后的问题字典 {corpus_name: [questions]}
        max_concurrent: 最大并发数（建议 1-3）
    """
    semaphore = asyncio.Semaphore(max_concurrent)
    
    async def process_with_limit(corpus_item):
        async with semaphore:
            corpus_name = corpus_item["corpus_name"]
            rag = await initialize_rag(corpus_name)
            
            # 构建索引
            await rag.ainsert(corpus_item["context"])
            
            # 查询问题
            corpus_questions = questions.get(corpus_name, [])
            results = await rag.abatch_query(corpus_questions)
            
            return results
    
    # 限制并发启动的任务数
    tasks = [process_with_limit(item) for item in corpus_data]
    all_results = await asyncio.gather(*tasks, return_exceptions=True)
    
    return all_results
```
