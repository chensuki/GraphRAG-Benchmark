## 数据返回规范

### 1. 查询结果对象

推荐定义一个标准的查询结果对象：

```python
from dataclasses import dataclass
from typing import List, Optional, Dict, Any

@dataclass
class QueryResult:
    """标准查询结果对象"""
    
    # 必需字段
    answer: str  # 生成的答案
    context: List[str]  # 检索到的上下文列表
    
    # 可选字段
    entities: Optional[List[Dict[str, Any]]] = None  # 实体列表
    relationships: Optional[List[Dict[str, Any]]] = None  # 关系列表
    chunks: Optional[List[Dict[str, Any]]] = None  # 文本块列表
    passages: Optional[List[Dict[str, Any]]] = None  # 段落列表
    sentences: Optional[List[Dict[str, Any]]] = None  # 句子列表
    
    # 元数据
    metadata: Optional[Dict[str, Any]] = None  # 其他元数据
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典格式"""
        return {
            "answer": self.answer,
            "context": self.context,
            "entities": self.entities,
            "relationships": self.relationships,
            "chunks": self.chunks,
            "passages": self.passages,
            "sentences": self.sentences,
            "metadata": self.metadata
        }
    
    def get_context_list(self) -> List[str]:
        """获取上下文列表（确保返回字符串列表）"""
        if isinstance(self.context, list):
            return [str(c) for c in self.context]
        elif isinstance(self.context, str):
            return [self.context] if self.context else []
        else:
            return []
```

### 1. LightRAG实现

```python
# LightRAG的实现
from lightrag import LightRAG

class LightRAGAdapter:
    def __init__(self, working_dir, llm_model_func, embedding_func, **kwargs):
        self.rag = LightRAG(
            working_dir=working_dir,
            llm_model_func=llm_model_func,
            embedding_func=embedding_func,
            **kwargs
        )
    
    async def build_index(self, content, **kwargs):
        """构建索引"""
        await self.rag.ainsert(content)
    
    async def aquery(self, question, top_k=10, **kwargs):
        """查询并返回答案"""
        from lightrag import QueryParam
        param = QueryParam(mode="hybrid", top_k=top_k)
        result = await self.rag.aquery(question, param=param)
        return result
    
    async def aquery_full(self, question, top_k=10, **kwargs):
        """完整查询"""
        from lightrag import QueryParam
        param = QueryParam(mode="hybrid", top_k=top_k)
        
        # 获取答案
        answer = await self.rag.aquery(question, param=param)
        
        # 获取上下文数据
        query_data = await self.rag.aquery_data(question, param=param)
        
        # 提取上下文
        context = []
        if query_data and "data" in query_data:
            chunks = query_data["data"].get("chunks", [])
            for chunk in chunks:
                if isinstance(chunk, dict) and "content" in chunk:
                    context.append(chunk["content"])
                elif isinstance(chunk, str):
                    context.append(chunk)
        
        return QueryResult(
            answer=str(answer),
            context=context,
            entities=query_data["data"].get("entities") if query_data else None,
            relationships=query_data["data"].get("relationships") if query_data else None
        )
```

### 2. ClearRAG实现

```python
# ClearRAG的实现
from clearrag import ClearRAG

class ClearRAGAdapter:
    def __init__(self, config_path=None, **kwargs):
        self.rag = ClearRAG(config_path=config_path, **kwargs)
    
    async def build_index(self, content, **kwargs):
        """构建索引"""
        await self.rag.abuild(content=content, source_type="text")
    
    async def aquery_full(
        self, 
        query, 
        top_k_sentences=10, 
        top_k_passages=10,
        enable_passage_retrieval=True,
        generate_answer=True,
        **kwargs
    ):
        """完整查询"""
        result = await self.rag.aquery_full(
            query=query,
            top_k_sentences=top_k_sentences,
            top_k_passages=top_k_passages,
            enable_passage_retrieval=enable_passage_retrieval,
            generate_answer=generate_answer
        )
        
        # 提取上下文
        context = result.extract_context_list(
            top_k_sentences=top_k_sentences,
            top_k_passages=top_k_passages
        )
        
        return QueryResult(
            answer=result.natural_language_answer or "",
            context=context if isinstance(context, list) else [context],
            metadata=result.metadata if hasattr(result, 'metadata') else None
        )
    
    def close(self):
        """清理资源"""
        if hasattr(self.rag, 'close'):
            self.rag.close()
```

```python
# HippoRAG的实现
from hipporag import HippoRAG

class HippoRAGAdapter:
    def __init__(self, save_dir, llm_model_name, embedding_model_name, **kwargs):
        self.rag = HippoRAG(
            save_dir=save_dir,
            llm_model_name=llm_model_name,
            embedding_model_name=embedding_model_name,
            **kwargs
        )
    
    def build_index(self, docs, **kwargs):
        """构建索引"""
        self.rag.index(docs=docs)
    
    def query(self, queries, top_k=10, **kwargs):
        """查询"""
        results = self.rag.rag_qa(queries=queries)
        return [r.to_dict() for r in results]
    
    def get_context(self, question, top_k=10, **kwargs):
        """获取上下文"""
        retrieval_results = self.rag.retrieve(
            queries=[question], 
            num_to_retrieve=top_k
        )
        
        contexts = []
        for result in retrieval_results:
            if hasattr(result, 'docs'):
                contexts.extend(result.docs)
        
        return contexts
```

### 5. 结果缓存（可选）

```python
from functools import lru_cache
import hashlib

def get_cache_key(question: str, **kwargs) -> str:
    """生成缓存键"""
    key_data = f"{question}_{sorted(kwargs.items())}"
    return hashlib.md5(key_data.encode()).hexdigest()

class CachedRAGFramework:
    def __init__(self, rag_instance, cache_size=100):
        self.rag = rag_instance
        self.cache = {}
        self.cache_size = cache_size
    
    async def aquery(self, question, **kwargs):
        """带缓存的查询"""
        cache_key = get_cache_key(question, **kwargs)
        
        if cache_key in self.cache:
            logger.debug(f"使用缓存结果: {cache_key}")
            return self.cache[cache_key]
        
        result = await self.rag.aquery(question, **kwargs)
        
        # 添加到缓存
        if len(self.cache) >= self.cache_size:
            # 简单的LRU：删除第一个
            first_key = next(iter(self.cache))
            del self.cache[first_key]
        
        self.cache[cache_key] = result
        return result
```

### 6. 批量查询支持

```python
async def abatch_query(
    self, 
    questions: List[str], 
    top_k: int = 10,
    **kwargs
) -> List[QueryResult]:
    """
    批量查询
    
    Args:
        questions: 问题列表
        top_k: 每个问题返回的上下文数量
        **kwargs: 其他参数
        
    Returns:
        查询结果列表
    """
    tasks = [self.aquery(q, top_k=top_k, **kwargs) for q in questions]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # 处理异常
    processed_results = []
    for i, result in enumerate(results):
        if isinstance(result, Exception):
            logger.error(f"问题 {i} 查询失败: {result}")
            processed_results.append(QueryResult(
                answer=f"查询失败: {str(result)}",
                context=[]
            ))
        else:
            processed_results.append(result)
    
    return processed_results
```

## 总结

实现一个GraphRAG框架接入Benchmark项目，核心是：

1. **实现必需接口**：初始化、索引构建、查询、获取上下文
2. **返回标准格式**：确保上下文是字符串列表，答案和字段类型正确
3. **支持异步操作**：提高性能和并发能力
4. **完善的错误处理**：确保异常情况下也能返回标准格式
5. **资源管理**：正确清理资源，避免内存泄漏

遵循本规范，您的框架将能够无缝接入GraphRAG-Benchmark项目，并与其他框架进行公平比较。
