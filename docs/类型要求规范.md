```python
def __init__(
    self,
    working_dir: str,
    llm_model_name: str,
    embedding_model_name: str,
```

| top_k | int | | 检索数量，默认10 |
| chunk_size | int | | 文本块大小，默认256 |
| chunk_overlap | int | | 文本块重叠，默认32 |
| temperature | float | | 温度参数，默认0.0 |

## 返回数据类型要求

### 1. 查询结果（QueryResult）

推荐使用dataclass定义标准返回类型：

```python
from dataclasses import dataclass
from typing import List, Optional, Dict, Any

@dataclass
class QueryResult:
    """标准查询结果类型"""
    
    # 必需字段
    answer: str                      # 字符串，必需
    context: List[str]               # 字符串列表，必需
    
    # 可选字段
    entities: Optional[List[Dict[str, Any]]] = None
    relationships: Optional[List[Dict[str, Any]]] = None
    chunks: Optional[List[Dict[str, Any]]] = None
    passages: Optional[List[Dict[str, Any]]] = None
    sentences: Optional[List[Dict[str, Any]]] = None
    metadata: Optional[Dict[str, Any]] = None
```

**类型要求详细说明**：

#### answer字段
```python
# ✅ 正确
answer: str = "这是答案文本"

# ❌ 错误
answer: List[str] = ["答案"]      # 不能是列表
answer: Dict = {"text": "答案"}   # 不能是字典
answer: None                      # 不能是None（可以用空字符串）
```

**必须**：字符串类型，即使是错误消息也要是字符串

#### context字段
```python
# ✅ 正确
context: List[str] = ["片段1", "片段2", "片段3"]  # 字符串列表
context: List[str] = ["单个片段"]                # 单个片段也要在列表中
context: List[str] = []                          # 空列表表示无上下文

# ❌ 错误
context: str = "单个片段"             # 不能是字符串
context: List[Dict] = [{"content": "片段"}]  # 不能是字典列表
context: None                        # 不能是None（用空列表）
```

**必须**：字符串列表，即使只有一个片段也要放在列表中

### 2. 字典格式的返回（替代方案）

如果不使用QueryResult对象，可以返回字典：

```python
def query(self, question: str, **kwargs) -> Dict[str, Any]:
    """返回字典格式"""
    return {
        "answer": str,           # 必需，字符串
        "context": List[str],    # 必需，字符串列表
        "entities": Optional[List[Dict]],    # 可选
        "relationships": Optional[List[Dict]], # 可选
        "metadata": Optional[Dict]            # 可选
    }
```

**关键要求**：
- `answer` 必须是字符串
- `context` 必须是字符串列表

### 3. 仅返回答案（简化版本）

某些框架可能只返回答案字符串：

```python
def query(self, question: str, **kwargs) -> str:
    """仅返回答案字符串"""
    return "这是答案"
```

**注意**：这种方式需要额外的方法获取上下文：

```python
def get_context(self, question: str, **kwargs) -> List[str]:
    """获取上下文列表"""
    return ["片段1", "片段2"]
```

### 1. 检索评估（retrieval_eval.py）

评估脚本期望的数据格式：

```python
# 输入数据格式（predictions_*.json）
[
    {
        "id": str,                           # 必需，字符串
        "question": str,                     # 必需，字符串
        "context": List[str],               # 必需，字符串列表
        "evidence": str,                     # 必需，字符串
        "question_type": str                 # 必需，字符串
    }
]
```

**评估脚本处理逻辑**：

```python
# 从评估脚本中可以看到：
context = item.get('context', [])
if isinstance(context, str):
    context_list = [context] if context else []
elif isinstance(context, list):
    context_list = context
else:
    context_list = []
```

**重要**：评估脚本会处理以下情况：
- 如果`context`是字符串 → 转换为单元素列表

**最佳实践**：始终返回列表类型，避免评估脚本进行类型转换。

## 类型验证和转换

### 1. 参数验证函数

```python
from typing import Optional

def validate_string(value: Optional[str], param_name: str, allow_none: bool = True) -> Optional[str]:
    """验证字符串参数"""
    if value is None:
        if not allow_none:
            raise ValueError(f"{param_name}不能为None")
        return value
    
    if not isinstance(value, str):
        raise TypeError(f"{param_name}必须是字符串类型，当前类型: {type(value)}")
    
    return value

def validate_int(value: Optional[int], param_name: str, min_val: int = 1, max_val: int = 1000) -> int:
    """验证整数参数"""
    if value is None:
        raise ValueError(f"{param_name}不能为None")
    
    if not isinstance(value, int):
        raise TypeError(f"{param_name}必须是整数类型，当前类型: {type(value)}")
    
    if value < min_val or value > max_val:
        raise ValueError(f"{param_name}必须在{min_val}到{max_val}之间，当前值: {value}")
    
    return value

def validate_context(context: Any) -> List[str]:
    """验证并转换context为列表格式"""
    if context is None:
        return []
    
    if isinstance(context, str):
        return [context] if context else []
    
    if isinstance(context, list):
        # 确保列表中的每个元素都是字符串
        return [str(item) if not isinstance(item, str) else item for item in context]
    
    # 其他类型转换为字符串列表
    return [str(context)]
```

### 2. 在框架中使用验证

```python
class MyRAGFramework:
    def __init__(
        self,
        working_dir: str,
        llm_model_name: str,
        embedding_model_name: str,
        **kwargs
    ):
        # 验证必需参数
        self.working_dir = validate_string(working_dir, "working_dir", allow_none=False)
        self.llm_model_name = validate_string(llm_model_name, "llm_model_name", allow_none=False)
        self.embedding_model_name = validate_string(embedding_model_name, "embedding_model_name", allow_none=False)
        
        # 验证可选参数
        self.top_k = validate_int(kwargs.get('top_k', 10), "top_k", min_val=1, max_val=100)
        
    async def aquery(self, question: str, top_k: Optional[int] = None, **kwargs) -> QueryResult:
        """查询方法，带类型验证"""
        # 验证输入参数
        question = validate_string(question, "question", allow_none=False)
        if top_k is not None:
            top_k = validate_int(top_k, "top_k", min_val=1, max_val=100)
        else:
            top_k = self.top_k
        
        # 执行查询...
        answer, context = await self._internal_query(question, top_k)
        
        # 验证并转换输出
        context = validate_context(context)
        answer = str(answer) if answer is not None else ""
        
        return QueryResult(
            answer=answer,
            context=context
        )
```

### 3. 类型转换工具

```python
def ensure_string(value: Any) -> str:
    """确保返回字符串"""
    if value is None:
        return ""
    if isinstance(value, str):
        return value
    return str(value)

def ensure_string_list(value: Any) -> List[str]:
    """确保返回字符串列表"""
    if value is None:
        return []
    if isinstance(value, str):
        return [value] if value else []
    if isinstance(value, list):
        return [ensure_string(item) for item in value]
    return [str(value)]

def ensure_dict(value: Any) -> Dict[str, Any]:
    """确保返回字典"""
    if value is None:
        return {}
    if isinstance(value, dict):
        return value
    return {"value": value}
```

## 常见类型错误和解决方案

### 1. context字段类型错误

**错误示例**：
```python
# ❌ 错误：context是字符串而不是列表
result = {
    "answer": "答案",
    "context": "单个上下文片段"  # 错误！
}
```

**解决方案**：
```python
# ✅ 正确：context必须是列表
result = {
    "answer": "答案",
    "context": ["单个上下文片段"]  # 正确！
}

# 或使用验证函数
result = {
    "answer": "答案",
    "context": ensure_string_list(context_value)  # 自动转换
}
```

### 2. answer字段类型错误

**错误示例**：
```python
# ❌ 错误：answer是None
result = {
    "answer": None,  # 错误！
    "context": ["片段"]
}

# ❌ 错误：answer是字典
result = {
    "answer": {"text": "答案"},  # 错误！
    "context": ["片段"]
}
```

**解决方案**：
```python
# ✅ 正确：answer必须是字符串
result = {
    "answer": "答案文本",  # 正确！
    "context": ["片段"]
}

# 或处理None情况
answer = query_result.get("answer")
result = {
    "answer": answer if answer else "无法生成答案",  # 正确！
    "context": ["片段"]
}
```

### 3. question参数类型错误

**错误示例**：
```python
# ❌ 错误：question是列表
result = rag.query(["问题1", "问题2"])  # 错误！

# ❌ 错误：question是字典
result = rag.query({"text": "问题"})  # 错误！
```

**解决方案**：
```python
# ✅ 正确：question必须是字符串
result = rag.query("这是一个问题")  # 正确！

# 批量查询需要循环处理
questions = ["问题1", "问题2"]
results = []
for q in questions:
    result = rag.query(q)  # 正确！
    results.append(result)
```

### 4. top_k参数类型错误

**错误示例**：
```python
# ❌ 错误：top_k是字符串
result = rag.query("问题", top_k="10")  # 错误！

# ❌ 错误：top_k是浮点数
result = rag.query("问题", top_k=10.0)  # 错误！

# ❌ 错误：top_k是负数
result = rag.query("问题", top_k=-5)  # 错误！
```

**解决方案**：
```python
# ✅ 正确：top_k必须是整数
result = rag.query("问题", top_k=10)  # 正确！

# 或使用验证
top_k = int(top_k_value) if isinstance(top_k_value, (int, float)) else 10
top_k = max(1, min(100, top_k))  # 限制范围
result = rag.query("问题", top_k=top_k)  # 正确！
```

### 5. content参数类型错误（索引构建）

**错误示例**：
```python
# ❌ 错误：content是列表
rag.build_index(["段落1", "段落2"])  # 错误！

# ❌ 错误：content是字典
rag.build_index({"text": "内容"})  # 错误！
```

**解决方案**：
```python
# ✅ 正确：content必须是字符串
rag.build_index("完整的文本内容...")  # 正确！

# 如果有多个段落，需要拼接
texts = ["段落1", "段落2", "段落3"]
rag.build_index("\n\n".join(texts))  # 正确！
```

## 总结

### 必须的类型要求

#### 传入参数
1. **字符串参数**：question、content、working_dir、model_name等
2. **整数参数**：top_k、chunk_size、chunk_overlap等
3. **可选参数**：使用Optional[T]类型提示

#### 返回数据
1. **answer字段**：必须是字符串（str）
2. **context字段**：必须是字符串列表（List[str]）
3. **所有字段**：必须存在，类型正确

### 核心原则

1. **类型一致性**：传入参数和返回数据的类型必须一致
2. **列表优先**：context字段始终使用列表格式
3. **字符串安全**：所有字符串字段确保不为None
4. **参数验证**：在接口入口处验证参数类型
5. **错误处理**：类型错误时返回安全的默认值

遵循本规范，可以确保您的框架正确接入GraphRAG-Benchmark项目，并与评估脚本无缝协作。
